class Solution {
    public int solution(int n, int m, int[] section) {
        int answer = 0;
        boolean[] wall = new boolean[n];

        for (int i : section) {
            wall[i-1] = true;
        }

        for (int i = 0; i < wall.length; i++) {
            if(wall[i]){
                answer++;
                i += m - 1;
            }
        }
        return answer;
    }
}
