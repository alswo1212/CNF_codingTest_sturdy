class Solution {
    final int divNum = 1234567;
    public long solution(int n) {
        if(n == 1){
            return 1;
        }
        long[] dp = new long[n+1];
        dp[1] = 1; dp[2] = 2;

        for (int i = 3; i < dp.length; i++) {
            dp[i] = (dp[i-1] + dp[i-2]) % divNum;
        }
        return dp[n];
    }
}
